// ==UserScript==
// @name         Smart MAX Blocker
// @namespace    http://tampermonkey.net
// @version      3.1
// @description  Блокирует только мессенджер MAX, не трогая другие сайты
// @author       Assistant
// @match        *://*.max.ru/*
// @match        *://max.ru/*
// @match        *://web.max.ru/*
// @match        *://messenger.max.ru/*
// @run-at       document-start
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // 1. Немедленная остановка загрузки
    window.stop();

    // 2. Очистка страницы
    document.documentElement.innerHTML = '';

    // 3. Редирект на пустую страницу, чтобы не висел белый экран
    window.location.replace("about:blank");

    console.log("MAX Messenger заблокирован скриптом.");
})();
